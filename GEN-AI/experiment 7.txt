from transformers import pipeline

summarizer = pipeline("summarization", model="sshleifer/distilbart-cnn-12-6")

text = """
Artificial Intelligence (AI) is a rapidly advancing field that aims to create machines capable of human-like thinking.
AI is used in various industries, from healthcare to finance, improving efficiency and accuracy.
"""

result = summarizer(text, max_length=100, min_length=20, do_sample=False)

print(result[0]['summary_text'])
